\begin{equation}
  \begin{aligned}
    \tau_{xx} 
    &=  -P + \lambda\bigg\{
      \pdv{u}{x} + \pdv{v}{r} + \frac{1}{r}\pdv{w}{\phi} + \frac{v}{r}
    \bigg\} \\
    \tau_{rr} 
    &= = -P +  \lambda\bigg\{
      \pdv{u}{x} + \pdv{v}{r} + \frac{1}{r}\pdv{w}{\phi} + \frac{v}{r}
      + 2\mu\pdv{v}{r} 
    \bigg\} \\
    \tau_{\phi\phi} 
    &= - P +  \lambda\bigg\{
      \pdv{u}{x} + \pdv{v}{r} + \frac{1}{r}\pdv{w}{\phi} + \frac{v}{r}
      + \frac{2\mu}{r}\pdv{w}{\phi} + \frac{2 v}{r} \bigg\} \\
    \tau_{xr}
    &= \mu\bigg\{\pdv{u}{x} + \pdv{v}{r}\bigg\} \\
    \tau_{x\phi} 
    &= \mu\bigg\{\pdv{w}{x} + \frac{1}{r}\pdv{u}{\phi}\bigg\} \\
    \tau_{r\phi}
    &= \mu\bigg\{\pdv{w}{r} + \frac{1}{r}\pdv{v}{\phi} - \frac{w}{r}
    \bigg\}
  \end{aligned}
\end{equation}
